def destroy
  @course = Course.find(params[:id])
  if can? :delete, @course
    @course.destroy
    flash[:notice] = 'Course deleted successfully'
  else
    flash[:notice] = 'Only admin can delete courses'
  end
    redirect_to '/courses'
end
